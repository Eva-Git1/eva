1    import android.os.Bundle
2    import androidx.appcompat.app.AppCompatActivity
3    import androidx.fragment.app.Fragment
4    import com.google.android.material.bottomnavigation.BottomNavigationView
5    
6    class MainActivity : AppCompatActivity(), MainActivityInterface {
7    
8        override fun onCreate(savedInstanceState: Bundle?) {
9            super.onCreate(savedInstanceState)
10           setContentView(R.layout.activity_main)
11   
12           // Configure le menu de navigation
13           setupBottomNavigation()
14       }
15   
16       override fun setupBottomNavigation() {
17           val bottomNavigation: BottomNavigationView = findViewById(R.id.bottom_navigation)
18   
19           // Charger le fragment par défaut
20           loadFragment(CategoryFragment())
21   
22           // Définir un listener pour les éléments du menu
23           bottomNavigation.setOnNavigationItemSelectedListener { item ->
24               val fragment: Fragment = when (item.itemId) {
25                   R.id.nav_category -> CategoryFragment()
26                   R.id.nav_video -> VideoFragment()
27                   R.id.nav_new_sales -> NewSalesFragment()
28                   else -> CategoryFragment()
29               }
30               loadFragment(fragment)
31               true
32           }
33       }
34   
35       override fun loadFragment(fragment: Fragment) {
36           supportFragmentManager.beginTransaction()
37               .replace(R.id.fragment_container, fragment)
38               .commit()
39       }
40   }
